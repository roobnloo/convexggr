generate <- function(n, d, p, tB, mG, seed) {
  set.seed(seed)
  s <- data_generate(n, d, p, tB, mG)
  return(s)
}

failures <- 0
trials <- 10
s_list <- vector(mode = "list", length = trials)
for (i in seq_len(trials)) {
  s <- generate(n, p, q, tB, mG, i)
  if (is.null(s)) {
    failures <- failures + 1
    next
  }
  s_list[[i]] <- s
}
